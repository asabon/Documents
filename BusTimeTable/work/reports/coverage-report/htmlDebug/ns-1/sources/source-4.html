 @Composable
 fun MyApplication(timeTable: TimeTable, config: Config) {
     BusTimeTableTheme {
         val navController = rememberNavController()
         NavHost(navController = navController, startDestination = "main") {
             composable("main") {
                 MainScreen(navController, timeTable)

             composable("config") {

             }
         }
     }
 }

 @Composable
 fun MainScreen(navController: NavController, timeTable: TimeTable) {
     var direction by remember { mutableIntStateOf(timeTable.getDefaultDirection()) }
     var dateType by remember { mutableIntStateOf(timeTable.getDefaultDateType()) }
     var tempList = timeTable.getTimeTableItems3(direction, dateType)
     val listState = rememberLazyListState(initialFirstVisibleItemIndex = getIndex(tempList))
     val composableScope = rememberCoroutineScope()
     Column(
         modifier = Modifier
             .fillMaxSize(),
         verticalArrangement = Arrangement.Bottom
     ) {
         Row(
             modifier = Modifier
                 .fillMaxWidth(),
             horizontalArrangement = Arrangement.End,
             verticalAlignment = Alignment.CenterVertically
         ) {
             SelectDirection(direction) {
                 direction = it
                 composableScope.launch {
                     tempList = timeTable.getTimeTableItems3(direction, dateType)
                     listState.animateScrollToItem(getIndex(tempList), 0)
                 }
             }
             Icon(
                 painter = rememberVectorPainter(image = Icons.Default.Settings),
                 contentDescription = null,
                 tint = Color.Gray,
                 modifier = Modifier
                     .size(40.dp)
                     .clickable {
                         navController.navigate("config")
                     }
             )
         }
         SelectDateType(dateType) {
             dateType = it
             composableScope.launch {
                 tempList = timeTable.getTimeTableItems3(direction, dateType)
                 listState.animateScrollToItem(getIndex(tempList), 0)
             }
         }
         tempList = timeTable.getTimeTableItems3(direction, dateType)
         TimeTableView(
             listState,
             tempList
         )
         // 参考 : https://qiita.com/tsumuchan/items/3e88e8fcc78a2781df85
         // 参考 : https://qiita.com/takagimeow/items/b0e1e45561ddf8b03aac
         AndroidView(
             modifier = Modifier.fillMaxWidth(),
             factory = { context ->
                 val adView = AdView(context).apply {
                     setAdSize(AdSize.BANNER)
                     adUnitId = "ca-app-pub-3940256099942544/6300978111"
                     loadAd(AdRequest.Builder().build())
                 }
                 adView.rootView
             }
         )
     }
 }

 @Composable
 fun TimeTableView(listState: LazyListState, itemList: List<TimeTableItem>) {
     LazyColumn(
         state = listState
     ) {
         items(itemList) {
                 item -> TimeTableItemView(item)
         }
     }
 }
 
 @Composable
 fun TimeTableItemView(item: TimeTableItem) {

             .fillMaxWidth()
     ){

             modifier = Modifier
                 .align(Alignment.CenterVertically),

             text = item.system + " "

             Text(

                 text = item.position + " → " + item.to
             )

             modifier = Modifier
                 .fillMaxWidth(),
             fontSize = 35.sp,
             textAlign = TextAlign.Right,

         )
     }
 }
