 @Composable
 fun SpotSelector(config: Config, initialValue: String, onClick: (String)->Unit = {}) {
     var expanded by remember { mutableStateOf(false) }
     val selectedOptionText : MutableState<String> = remember { mutableStateOf(initialValue) }
     Box(
         modifier = Modifier
             .size(150.dp, 30.dp)
             .border(BorderStroke(1.dp, Color.LightGray), RoundedCornerShape(4.dp))
             .clickable {
                 expanded = !expanded
             }
     ) {
         Text(
             text = selectedOptionText.value,
             modifier = Modifier
                 .padding(start = 5.dp)
         )
         Icon(
             Icons.Filled.ArrowDropDown,
             "contentDescription",
             modifier = Modifier
                 .padding(start = 130.dp)
         )
         DropdownMenu(
             expanded = expanded,
             onDismissRequest = { expanded = false }
         ) {
             config.getSpotList().forEach() {
                 DropdownMenuItem(
                     text = { Text(text = it) },
                     onClick = {
                         onClick(it)
                         selectedOptionText.value = it
                         expanded = false
                     }
                 )
             }
         }
     }
 }
