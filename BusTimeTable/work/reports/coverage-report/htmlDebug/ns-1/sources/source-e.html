 /**
  * 第一引数の文字列が、第二引数の文字列リストに含まれているかを確認する
  * */
 fun isContain(target: String, list: List<String>): Boolean {
     for (item in list) {
         if (target == item) {
             return true
         }
     }
     return false
 }
 
 /**
  * 第一引数の文字列リストの中のどれか一つ以上が第二引数の文字列リストに含まれているかを確認する

 fun isContains(targets: List<String>, list: List<String>): Boolean {
     for (target in targets) {

     }
