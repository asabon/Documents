 @Composable
 fun StationView(modifier: Modifier, listState: LazyListState, stationList: List<String>) {
     LazyColumn(
         state = listState,
         modifier = modifier
     ) {
         items(stationList) {
                 item -> Text(text = item)
         }
     }
 }
