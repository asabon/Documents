 package net.asabon.bustimetable
 
 import android.content.Context
 import com.fasterxml.jackson.module.kotlin.jacksonObjectMapper
 import com.fasterxml.jackson.module.kotlin.readValue
 import java.io.File
 
 data class SpotInfo(
     val name: String,
     val busStops: MutableList<String>
 )
 
 class Config(context: Context?) {
 
     private val context = context
     private var spotInfos = mutableListOf<SpotInfo>()
 
     fun initialize() {
         spotInfos.clear()
         spotInfos.add(
             SpotInfo(
                 "家",
                 mutableListOf(
                     "町田総合高校前",
                     "木曽南団地",
                     "上宿",
                     "境川団地"
                 )
             )
         )
         spotInfos.add(
             SpotInfo(
                 "駅",
                 mutableListOf(
                     "町田バスセンター"
                 )
             )
         )
     }
 
     /**
      * 保存されているスポットリストを取得する
      */
     fun getSpotList(): List<String> {
         val spots = mutableListOf<String>()
         for (spotInfo in spotInfos) {
             spots.add(spotInfo.name)
         }
         return spots
     }
 
     fun getBusStopList(spotName: String): List<String> {
         for (spot in spotInfos) {
             if (spot.name == spotName) {
                 return spot.busStops
             }
         }
         return listOf("")
     }
 
     /**
      * スポットを新規登録する
      */
     fun addSpot(spot: String) {
         for (spotInfo in spotInfos) {
             if (spot == spotInfo.name) {
                 /* 重複 */
                 return
             }
         }
         spotInfos.add(SpotInfo(spot, mutableListOf<String>()))
     }
 
     /**
      * スポットを削除する
      */
     fun removeSpot(spot: String) {
         for (spotInfo in spotInfos) {
             if (spot == spotInfo.name) {
                 spotInfos.remove(spotInfo)
                 return
             }
         }
     }
 
     /**
      * スポットにバス停を登録する
      */
     fun addBusStop(spot: String, busStop: String) {
         for (spotInfo in spotInfos) {
             if (spot == spotInfo.name) {
                 spotInfo.busStops.add(busStop)
                 return
             }
         }
     }
 
     /**
      * スポットからバス停を削除する
      */
     fun removeBusStop(spot: String, busStop: String) {
         for (spotInfo in spotInfos) {
             if (spot == spotInfo.name) {
                 spotInfo.busStops.remove(busStop)
                 return
             }
         }
     }
 
     /**
      * 内部ストレージに保存した設定をすべて消去する
      */
     fun clear() {
         spotInfos.clear()
         if (context != null) {
             try {
                 File(context.filesDir, "spotInfos.json").delete()
             } catch (e: Exception) {
                 print(e)
             }
         }
     }
 
     /**
      * 内部ストレージに設定を保存する
      */
     fun save() {
         if (context != null) {
             try {
                 val mapper = jacksonObjectMapper()
                 mapper.writeValue(File(context.filesDir, "spotInfos.json"), spotInfos)
             } catch (e: Exception) {
                 print(e)
             }
         }
     }
 
     /**
      * 内部ストレージから設定を読み出す
      */
     fun load() {
         if (context != null) {
             try {
                 val file = File(context.filesDir, "spotInfos.json")
                 val jsonString = file.bufferedReader().use { it.readText() }
                 val mapper = jacksonObjectMapper()
                 spotInfos = mapper.readValue<List<SpotInfo>>(jsonString).toMutableList()
             } catch(e: Exception) {
                 println(e)
             }
         }
     }
 }
