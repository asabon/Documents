 @Composable
 fun SelectDirection(timeTable: TimeTable, direction: Int, onClick: (Int)->Unit = {}) {
     val directionIcon = listOf(Icons.Default.ArrowForward, Icons.Default.ArrowBack)
     Row{
         SpotSelector(
             timeTable = timeTable,
             initialValue = timeTable.getSpotA()
         ) {
             timeTable.setSpotA(it)
             timeTable.saveConfig()
             onClick(1)
         }
         Icon(
             painter = rememberVectorPainter(image = directionIcon[direction]),
             contentDescription = null,
             tint = Color.Black,
             modifier = Modifier
                 .size(50.dp)
                 .clickable {
                     if (timeTable.getDirection() == 0) {
                         timeTable.setDirection(1)
                     } else {
                         timeTable.setDirection(0)
                     }
                     timeTable.saveConfig()
                     onClick(0)
                 }
         )
         SpotSelector(
             timeTable = timeTable,
             initialValue = timeTable.getSpotB()
         ) {
             timeTable.setSpotB(it)
             timeTable.saveConfig()
             onClick(2)
         }
     }
 }

 @Composable
 fun TimeTableView(modifier: Modifier, listState: LazyListState, itemList: List<TimeTableItem>) {
