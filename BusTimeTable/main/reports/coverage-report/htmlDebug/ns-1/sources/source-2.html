 }
 
 /**
  * スポットを選択するドロップダウンメニュー
  *
  * @param timeTable TimeTable インスタンス
  * @param initialValue 初期値
  * @param onClick コールバック関数
  */
 @Composable
 fun SpotSelector(timeTable: TimeTable, initialValue: String, onClick: (String)->Unit = {}) {
     var expanded by remember { mutableStateOf(false) }
     val selectedOptionText : MutableState<String> = remember { mutableStateOf(initialValue) }
     Box(
         modifier = Modifier
             .size(150.dp, 50.dp)
             .border(BorderStroke(1.dp, Color.LightGray), RoundedCornerShape(4.dp))
             .clickable {
                 expanded = !expanded
             }
     ) {
         Text(
             text = selectedOptionText.value,
             modifier = Modifier
                 .padding(start = 5.dp)
                 .align(Alignment.Center),
             fontSize = 25.sp
         )
         Icon(
             Icons.Filled.ArrowDropDown,
             "contentDescription",
             modifier = Modifier
                 .padding(start = 130.dp)
         )
         DropdownMenu(
             expanded = expanded,
             onDismissRequest = { expanded = false }
         ) {
             timeTable.getSpotList().forEach() {
                 DropdownMenuItem(
                     text = {
                         Text(
                             text = it,
                             fontSize = 16.sp
                         )
                     },
                     onClick = {
                         onClick(it)
                         selectedOptionText.value = it
                         expanded = false
                     }
                 )
             }
         }
     }
 }
