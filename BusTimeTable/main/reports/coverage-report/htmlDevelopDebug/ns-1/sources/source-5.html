 @Composable
 fun BusStopListView(
     modifier: Modifier,
     listState: LazyListState,
     busStopList: List<String>,
     selectedBusStop: String,
     onClick: (String) -> Unit = {},
 ) {
     var selectedIndex by remember { mutableIntStateOf(-1) }
     LazyColumn(
         state = listState,
         modifier = modifier,
     ) {
         itemsIndexed(busStopList) { index, item ->
             Text(
                 text = item,
                 modifier = Modifier
                     .clickable {
                         selectedIndex = index
                         onClick(item)
                     }
                     .background(
                         color = if (item == selectedBusStop) {
                             Color.Gray
                         } else {
                             Color.Transparent
                         },
                     ),
             )
         }
     }
 }
