 package net.asabon.couponmanager
 
 import android.content.Context
 import android.util.Log
 import com.fasterxml.jackson.module.kotlin.jacksonObjectMapper
 import com.fasterxml.jackson.module.kotlin.readValue
 import java.io.File
 
 class CouponManager(context: Context?) {
 
     private val TAG = "CouponManager"
     private val datafile = "couponManagerData.json"
     private val mContext = context
     private var mCouponItemList: MutableList<CouponItem> = mutableListOf()
 
     /**
      * 登録済みクーポンリストを取得する
      */
     fun getItemList(): MutableList<CouponItem> {
         return mCouponItemList
     }
 
     /**
      * データを全消去する
      */
     fun clear() {
         mCouponItemList.clear()
         if (mContext != null) {
             try {
                 File(mContext.filesDir, datafile).delete()
             } catch (e: Exception) {
                 print(e)
                 Log.e(TAG, "$e")
             }
         }
         Log.d(TAG, "clear() end")
     }
 
     /**
      * データをロードする
      */
     fun load() {
         mCouponItemList.clear()
         if (mContext == null) {
             createSampleData()
         } else {
             try {
                 val file = File(mContext.filesDir, datafile)
                 if (file.exists()) {
                     val jsonString = file.bufferedReader().use { it.readText() }
                     val mapper = jacksonObjectMapper()
                     mCouponItemList = mapper.readValue(jsonString)
                 }
             } catch (e: Exception) {
                 print(e)
             }
         }
     }
 
     /**
      * データをセーブする
      */
     fun save() {
         if (mContext != null) {
             try {
                 val mapper = jacksonObjectMapper()
                 mapper.writeValue(File(mContext.filesDir, datafile), mCouponItemList)
             } catch (e: Exception) {
                 print(e)
             }
         }
     }
 
     /**
      * デバッグ用のサンプルデータを生成する
