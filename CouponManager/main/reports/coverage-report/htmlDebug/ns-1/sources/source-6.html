 package net.asabon.couponmanager
 
 import android.content.Context
 import com.fasterxml.jackson.module.kotlin.jacksonObjectMapper
 import com.fasterxml.jackson.module.kotlin.readValue
 import java.io.File
 
 class CouponItemList(
     private val fileName: String,
     private val context: Context,
 ) {
     private var itemList: MutableList<CouponItem> = mutableListOf()
 
     fun setList(newList: List<CouponItem>) {
         itemList = newList.toMutableList()
     }
 
     fun getList(): MutableList<CouponItem> {
         return itemList
     }
 
     fun clear() {
         itemList.clear()
         try {
             File(context.filesDir, fileName).delete()
         } catch (e: Exception) {
             print(e)
         }
     }
 
     fun load() {
         itemList.clear()
         try {
             val file = File(context.filesDir, fileName)
             if (file.exists()) {
                 val jsonString = file.bufferedReader().use { it.readText() }
                 val mapper = jacksonObjectMapper()
                 itemList = mapper.readValue(jsonString)
             }
         } catch (e: Exception) {
             print(e)
         }
     }
 
     fun loadFromJsonString(jsonString: String) {
         val mapper = jacksonObjectMapper()
         itemList.clear()
         itemList = mapper.readValue(jsonString)
     }
 
     fun save() {
         try {
             val mapper = jacksonObjectMapper()
             mapper.writeValue(File(context.filesDir, fileName), itemList)
         } catch (e: Exception) {
             print(e)
         }
     }
 }
