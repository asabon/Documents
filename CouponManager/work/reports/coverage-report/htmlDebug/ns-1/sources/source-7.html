 package net.asabon.couponmanager
 
 import android.content.Context
 import android.content.res.AssetManager
 
 class CouponManager(
     context: Context,
     private val assetManager: AssetManager,
 ) {
     private var useSample = false
     private val validDataFileName = "validData.json"
     private val invalidDataFileName = "invalidData.json"
     private var validItemList = CouponItemList(validDataFileName, context)
     private var invalidItemList = CouponItemList(invalidDataFileName, context)
 
     fun useSample(flag: Boolean) {
         useSample = flag
     }
 
     /**
      * 有効なクーポンリストを取得する
      */
     fun getValidItemList(): MutableList<CouponItem> {
         return validItemList.getList()
     }
 
     /**
      * 無効（期限切れ）のクーポンリストを取得する
      */
     fun getInvalidItemList(): MutableList<CouponItem> {
         return invalidItemList.getList()
     }
 
     /**
      * データを全消去する
      */
     fun clear() {
         validItemList.clear()
         invalidItemList.clear()
     }
 
     /**
      * データをロードする
      */
     fun load() {
         if (useSample) {
             readSampleData(assetManager)
         } else {
             validItemList.load()
             invalidItemList.load()
         }
     }
 
     /**
      * データをセーブする
      */
     fun save() {
         validItemList.save()
         invalidItemList.save()
     }
 
     /**
      * デバッグ用のサンプルデータを生成する
      */
     private fun readSampleData(assetManager: AssetManager) {
         val validDataStream = assetManager.open("sample/$validDataFileName")
         val validDataJsonString = validDataStream.bufferedReader().use { it.readText() }
         validItemList.loadFromJsonString(validDataJsonString)
 
         val invalidDataStream = assetManager.open("sample/$invalidDataFileName")
         val invalidDataJsonString = invalidDataStream.bufferedReader().use { it.readText() }
         invalidItemList.loadFromJsonString(invalidDataJsonString)
     }
 }
